                      function rowSumOddNumbers(n) {
                        let number = 1;
                        const numbers = [1];
                        let result = 0;

                        for (let i = 1; i < n; i += 1) {
                          for (let j = 0; j < i + 1; j += 1) {
                            number += 2;
                            numbers.push(number);
                          }
                        }

                        for (let i = numbers.length - n; i < numbers.length; i += 1) {
                          result += numbers[i];
                        }

                        return result;
                      }
